// check the status
$ git status

// initializing a git file in project folder
$ git init

// add origin
$ git remote add origin http://github.com/username/repo.git

//
$ git add <filename or . >
// after adding the filename should change from red to green

// commit locally with title "Some text"
$ git commit -m "Some text"

// push to master
$ git push -u origin master

// creating a new branching
$ git branch <branch name>

// change to a different branch
$ git checkout <branch name>
